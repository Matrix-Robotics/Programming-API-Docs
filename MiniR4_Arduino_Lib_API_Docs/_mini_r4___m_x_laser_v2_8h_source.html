36// Decode VCSEL (vertical cavity surface emitting laser) pulse period in PCLKs
37// from register value
38// based on VL53L0X_decode_vcsel_period()
39#define decodeVcselPeriod(reg_val) (((reg_val) + 1) << 1)
40
41// Encode VCSEL pulse period register value from period in PCLKs
42// based on VL53L0X_encode_vcsel_period()
43#define encodeVcselPeriod(period_pclks) (((period_pclks) >> 1) - 1)
44
45// Calculate macro period in *nanoseconds* from VCSEL period in PCLKs
46// based on VL53L0X_calc_macro_period_ps()
47// PLL_period_ps = 1655; macro_period_vclks = 2304
48#define calcMacroPeriod(vcsel_period_pclks) ((((uint32_t)2304 * (vcsel_period_pclks) * 1655) + 500) / 1000)
