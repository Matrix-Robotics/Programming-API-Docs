16bool MatrixColorV3::begin() {
17 // 检查传感器ID
18 _pWire->begin();
19 i2cMUXSelect();
20 _pWire->beginTransmission(TCS34725_ADDRESS);
21 _pWire->write(TCS34725_COMMAND_BIT | TCS34725_ID);
22 _pWire->endTransmission();
23 _pWire->requestFrom(TCS34725_ADDRESS, 1);
24 if (_pWire->read() != 0x44) { // TCS34725的默认ID值
25 return false;
26 }
27
28 // 配置集成时间和增益
29 writeRegister(TCS34725_ATIME, TCS34725_INTEGRATIONTIME_50MS);
30 writeRegister(TCS34725_CONTROL, TCS34725_GAIN_1X);
31
32 // 启用传感器
33 writeRegister(TCS34725_ENABLE, TCS34725_ENABLE_PON);
34 delay(3); // 启动延迟
35 writeRegister(TCS34725_ENABLE, TCS34725_ENABLE_PON | TCS34725_ENABLE_AEN);
36
37 return true; // 初始化成功
38}

56float MatrixColorV3::calcColorTemp(uint16_t r, uint16_t g, uint16_t b) {
57 if (r == 0 || g == 0 || b == 0) return 0; // 避免零值导致的问题
58
59 float X = (-0.14282 * r) + (1.54924 * g) + (-0.95641 * b);
60 float Y = (-0.32466 * r) + (1.57837 * g) + (-0.73191 * b);
61 float Z = (-0.68202 * r) + (0.77073 * g) + (0.56332 * b);
62
63 if (X + Y + Z == 0) return 0; // 避免除以零
64
65 float x = X / (X + Y + Z);
66 float y = Y / (X + Y + Z);
67
68 // CIE 1931 2° 标准观察者公式
69 float n = (x - 0.3320) / (0.1858 - y);
70 float colorTemp = (449.0 * pow(n, 3)) + (3525.0 * pow(n, 2)) + (6823.3 * n) + 5520.33;
71
72 return colorTemp;
73}

79void MatrixColorV3::RGB2HSV(uint16_t r, uint16_t g, uint16_t b, float &h, float &s, float &v) {
80 float rd = r / 255.0;
81 float gd = g / 255.0;
82 float bd = b / 255.0;
83
84 float maxVal = max(rd, max(gd, bd));
85 float minVal = min(rd, min(gd, bd));
86
87 v = maxVal;
88 float delta = maxVal - minVal;
89
90 if (delta < 0.00001) {
91 s = 0;
92 h = 0; // undefined, maybe nan?
93 return;
94 }
95 if (maxVal > 0.0) { // NOTE: if Max is == 0, this divide would cause a crash
96 s = (delta / maxVal);
97 } else {
98 // if max is 0, then r = g = b = 0
99 // s = 0, v is undefined
100 s = 0;
101 h = NAN; // its now undefined
102 return;
103 }
104 if (rd >= maxVal) h = (gd - bd) / delta;
105 else if (gd >= maxVal) h = 2.0 + (bd - rd) / delta;
106 else h = 4.0 + (rd - gd) / delta;
107 h *= 60. ;
108}
